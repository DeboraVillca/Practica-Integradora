const Tutorial =({indice,tutorial,activarTutorial,borrarTutorial})=>{

    return(
        <div className="tutorial">
           {tutorial.tipo}- {tutorial.nombre}
            <div>
                
                <button onClick={()=>activarTutorial(indice)}>activar</button>
                <button onClick={()=>borrarTutorial(indice)}>Borrar</button> 
                <a href={tutorial.url}>  Link Tutorial</a> 
            </div>
            
        </div>
    )
}
export default Tutorial
